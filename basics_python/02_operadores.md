### Operadores aritmeticos ###

# Operadores enteros
print(3 + 4)
print(3 - 4)
print(3 * 4)
print(3 / 4)
print(10 % 3)  # modulo
print(10 // 3)  # division entera
print(2 ** 3)  # potencia 
print(2 ** 3 + 3 - 7 / 1 //4)

# Operaciones con cadenas de texto
print("Hola " + "Python " + "¿que tal?")
print("Hola " + str(5))

# Operaciones mixtas
print("Hola " * 5)  # Repite Hola 5 veces
print("Hola " * (2 ** 3))  # Repite Hola 8 veces

my_float = 2.5 * 2
print("Hola " * int(my_float))  # Cuando haces una operacion entre dos numeros se comvierte en float
                                # la solucion a que comvertirala en integer int()
### Operadores Comparativos ###
print(3 > 4)
print(3 < 4)
print(3 >= 4)
print(4 <= 4)
print(3 == 4)
print(3 != 4)

# Operaciones con cadenas de texto
print("Hola" > "Python")
print("Hola" < "Python")
print("aaaa" >= "abaa")  # Ordenación alfabética por ASCII
print(len("aaaa") >= len("abaa"))  # Cuenta caracteres
print("Hola" <= "Python")
print("Hola" == "Hola")
print("Hola" != "Python")
print("last part")
### Operadores Lógicos ###
print(3 > 4 and "Hola" > "Python")
print(3 > 4 or "Hola" > "Python")
print(3 < 4 and "Hola" < "Python")
print(3 < 4 or "Hola" > "Python")
print(3 < 4 or ("Hola" > "Python" and 4 == 4))
print(not (3 > 4))
